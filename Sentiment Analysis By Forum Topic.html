# Reading in html from website
url =  'http://www.cafepharma.com/boards/'
client = ureq(url) # opens connection
webpage = client.read() # grabs the page
client.close() # closing the client

# Parsing data, and storing to database
page_soup = soup(webpage, "html.parser")

count=0

for p, q in zip(
        page_soup.find_all('div', class_="listBlock main"),
        page_soup.find_all('div', class_="listBlock lastPost")):
    
    # Assigning values
    post_id = re.sub("[^0-9]", "", p.find('a')['href'])
    title = p.find('a').text
    content= p.find('p').text
    user=q.find('a').text
    posttime=q.find('abbr').text
    
    # Assigning to dictionary
    record = {'_id':post_id,
                'title': title,
                'content':content,
                'user': user,
                'posttime':posttime}
    
    if posts.find({'_id' : post_id}).count() < 1: 
        posts.insert(record)
        count+=1

print(str(count) + ' records added')

# Helper functions and dictionaries
stop_words = stopwords.words('english')

def average_word_length(review):
    words = review.split()
    if len(words) != 0:        
        return(sum(len(word) for word in words)/len(words))
    else:
        return(0)
    
def datetime_convert(posttime):
    return(dt.datetime.strptime(
        posttime,'%b %d, %Y at %I:%M %p')
    )

# Imputed metrics on content values
df['word_count'] = df['content'].apply(lambda x: len(x.split()))
df['character_count'] = df['content'].apply(lambda x: len(x))
df['average_word_length'] = df['content'].apply(lambda x: average_word_length(x))
df['stopword_count'] = df['content'].apply(
    lambda x: len([word for word in x.split() if word.lower() in stop_words])
    )
df['stopword_proportion'] = df['stopword_count']/df['word_count']
df['posttime'] = df['posttime'].apply(
    lambda x:
        datetime_convert(x)
    )

df.describe().round(1)
